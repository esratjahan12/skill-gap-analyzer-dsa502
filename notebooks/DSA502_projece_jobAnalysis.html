def pick_first_existing(df, candidates):
    for c in candidates:
        if c in df.columns:
            return c
    return None

def label_role(title: str):
    if not isinstance(title, str):
        return None
    t = title.lower()
    if "data scientist" in t:
        return "Data Scientist"
    if "data analyst" in t:
        return "Data Analyst"
    return None

def normalize_skill_token(s: str):
    s = s.strip().lower()
    # normalize some common variants
    s = s.replace("powerbi", "power bi")
    s = s.replace("machine-learning", "machine learning")
    return s

def split_skills(skill_text: str):
    """Split a skills string into a list, robust to commas/semicolons/pipes/newlines."""
    if not isinstance(skill_text, str) or not skill_text.strip():
        return []
    parts = re.split(r"[,\n;|/]+", skill_text)
    parts = [normalize_skill_token(p) for p in parts if p and p.strip()]
    # remove very short noise
    return [p for p in parts if len(p) >= 2]

from collections import Counter

def simple_tokenize(s: str):
    s = s.lower()
    s = re.sub(r"[^a-z0-9+\-# ]", " ", s)  # keep things like c++, c#, etc.
    toks = [t for t in s.split() if len(t) >= 2]
    return toks

from sklearn.model_selection import train_test_split
from sklearn.pipeline import Pipeline
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import classification_report, roc_auc_score, confusion_matrix

# Target
y = df_role["role_label"]
X = df_role["text"].fillna("")

# Train/val split (stratified to preserve class balance)
X_train, X_val, y_train, y_val = train_test_split(
    X, y, test_size=0.2, random_state=RANDOM_SEED, stratify=y
)

baseline = Pipeline(steps=[
    ("tfidf", TfidfVectorizer(
        stop_words="english",
        min_df=3,
        ngram_range=(1,2)
    )),
    ("clf", LogisticRegression(max_iter=2000))
])

baseline.fit(X_train, y_train)

pred = baseline.predict(X_val)
print(classification_report(y_val, pred))

# ROC-AUC (binary)
proba = baseline.predict_proba(X_val)
classes = list(baseline.named_steps["clf"].classes_)
print("Classes:", classes)

# pick "Data Scientist" as positive class if present
if "Data Scientist" in classes:
    pos_idx = classes.index("Data Scientist")
    auc = roc_auc_score((y_val == "Data Scientist").astype(int), proba[:, pos_idx])
    print("ROC-AUC (Data Scientist as positive):", auc)
